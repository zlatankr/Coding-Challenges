set @N = (select count(*) from STATION);
set @one_med = (select ceil(count(*)/2) from STATION);
set @two_med = (select count(*)/2 from STATION);

select
    case
        when @N = 1
        then (
            select round(LAT_N, 4)
            
            from STATION
            order by LAT_N
            LIMIT @one_med, 1
        
        when @N = 0
        then (
            select round(avg(LAT_N), 4)
            
            from STATION
            order by LAT_N
            LIMIT @two_med, 2
            
        else 'error'
        end
from STATION
